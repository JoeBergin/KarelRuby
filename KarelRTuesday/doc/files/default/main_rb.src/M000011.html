def task()
  world = Robota::World
  world.read_world("../worlds/stair_world.txt")
  
  karel = StairSweeper.new(1, 1, EAST, 0, TRANSPARENT)
  karel.climb_stair()
  karel.pick_beeper()
  karel.climb_stair()
  karel.pick_beeper()
  karel.climb_stair()
  karel.pick_beeper()
  karel.turn_off()
  karel.display()
  world.show_world_with_robots(1, 1, 6, 6)
  karel.display()
  
end
