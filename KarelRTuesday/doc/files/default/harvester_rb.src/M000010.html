def task()
    world = RobotWorld.instance
    world.read_world("../worlds/fig3-2.kwld")
    world.show_world_with_robots()
    mark =  Harvester.new(2, 2, EAST, 0)
    mark.move()
    mark.harvest_two_rows()
    mark.position_for_next_harvest()
    mark.harvest_two_rows()
    mark.position_for_next_harvest()
    mark.harvest_two_rows()
    mark.move()
    mark.turn_off()
    world.show_world_with_robots()
end
