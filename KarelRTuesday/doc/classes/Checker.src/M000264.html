    def enough_to_carry?() # Facing north on 1st Street. Checker class
        10.times do
            if next_to_a_beeper?() 
                pick_beeper()
            else 
                empty_bag()
                return false
            end
        end
        # Found ten beepers. Put them on 2nd street.
        move()
        empty_bag()
        back_up()
        return true
    end
