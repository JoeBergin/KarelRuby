       def initialize(street, avenue, number, window)
            @street = street
            @avenue = avenue
            @number = number
            @scaler = lambda{|x,y| window.scale_to_pixels(x,y)} #i.e. the function itself
            @scale_factor = lambda{window.scale_factor}
            @canvas = window.canvas
            @tag = "b" + @@bNumber.to_s
            @@bNumber += 1
        end
