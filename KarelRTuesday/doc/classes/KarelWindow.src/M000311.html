  def set_size(streets)
    @_streets = streets
    @_walls.each do  |wall|
      @canvas.delete(wall)
    end
    make_streets_and_avenues
    make_boundary_walls
    label_streets_avenues
    (@_contents + @_robots).each do |item|
      item.move_scale
    end
  end
