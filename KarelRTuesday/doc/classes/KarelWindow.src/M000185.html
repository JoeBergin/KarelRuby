     def delete_beeper(beeperLocation) # removes ALL beepers at this location
       # sleep 0.5
       beeper = @@Beepers[beeperLocation]
       if beeper
         beeper.delete_all
         @@Beepers.delete(beeperLocation)
         @_contents.delete(beeper)
       end
     end
