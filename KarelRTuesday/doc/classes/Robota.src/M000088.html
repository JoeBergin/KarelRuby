  def to_s
    if running?
      running = ", running"
    else
      running = ", not running"
    end
    howMany = @beepers
    howMany = "infinitely many" if howMany == INFINITY
    return "#{self.class} with ID #@ID at (#@street, #@avenue) facing #@direction with " + howMany.to_s + " beepers" + running
  end
