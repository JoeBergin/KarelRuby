  def neighbors_of(robot, street, avenue)
    result = []
    robot.assert_at(street, avenue)
    @Robots.keys.each do |anyRobot|
      state = @Robots[anyRobot]
      x, y = state[0], state[1]
      result << anyRobot if anyRobot != robot && x == street && y == avenue
    end
    return result
  end
