  def rotate
    @direction = $directions[($directions.index(@direction) + 1) %4]
    delete_all 
    @image = @imageMap[@direction]
     x, y = scaler(@street, @avenue)
     if @canvas
       @image.place(:x => x , :y => y)
       if @color == nil
         @bg_color = :white
         if $world.beepers_at?(@street, @avenue)
           @bg_color = :black
         end
           @image.configure(:bg => @bg_color)
       end
     end    
  end
