  def move(amount = 1)
     dx, dy = $moveParameters[@direction]
     @street -= dy*amount
     @avenue += dx*amount
     x, y = scaler(@street, @avenue)
     if @canvas
       @canvas.update
       @image.place(:x => x , :y => y)
       if @color == nil
         @bg_color = :white
         if $world.beepers_at?(@street, @avenue)
           @bg_color = :black
         end
           @image.configure(:bg => @bg_color)
       end
     end
  end
