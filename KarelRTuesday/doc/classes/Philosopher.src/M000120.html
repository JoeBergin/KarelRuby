 def get_forks
    turn_left()
    move()
    while ! any_beepers_in_beeper_bag?()
      while not next_to_a_beeper?()
        spin()
      end
      pick_beeper()
    end
    turn_around()
    move()
    put_beeper()
    move()
    while ! any_beepers_in_beeper_bag?()
      while not next_to_a_beeper?()
        spin()
      end
      pick_beeper()
    end
    turn_around()
    move()
    put_beeper()
    turn_right()
    # show_state("Eat ")
  end
