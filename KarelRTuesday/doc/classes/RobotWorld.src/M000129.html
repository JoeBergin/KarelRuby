  def place_beepers(street, avenue, howMany, byRobot = false)
    key = [street, avenue]
    if howMany == Robota::INFINITY || @Beepers[key] == Robota::INFINITY
      @Beepers[key] = Robota::INFINITY
    else
      @Beepers[key] += howMany   
    end
    nil
  end
